string(REGEX MATCH "([0-9.]+)$" MODO_VERSION "${CMAKE_CURRENT_SOURCE_DIR}")
set(MESHSYNC_CLIENT_MODO "MeshSyncClientModo${MODO_VERSION}")
find_package(Modo${MODO_VERSION})

file(GLOB sources *.cpp *.h)
add_library(${MESHSYNC_CLIENT_MODO} SHARED ${sources})
add_dependencies(${MESHSYNC_CLIENT_MODO} MeshUtils MeshSync)
target_include_directories(${MESHSYNC_CLIENT_MODO} PUBLIC "${CMAKE_SOURCE_DIR}" "${MODO${MODO_VERSION}_INCLUDE_DIR}")
target_link_libraries(${MESHSYNC_CLIENT_MODO} MeshSync MeshUtils ${Poco_LIBRARIES} ${MODO${MODO_VERSION}_LIBRARIES})
if(LINUX)
    target_link_libraries(${MESHSYNC_CLIENT_MODO} pthread "-Wl,--no-undefined")
elseif(MAC)
    set_target_properties(${MESHSYNC_CLIENT_MODO} PROPERTIES SUFFIX ".bundle")
endif()
set_target_properties(${MESHSYNC_CLIENT_MODO} PROPERTIES PREFIX "")
set_target_properties(${MESHSYNC_CLIENT_MODO} PROPERTIES OUTPUT_NAME "MeshSyncClientModo")

set(DIST_DIR "${CMAKE_INSTALL_PREFIX}/UnityMeshSync_Modo_${PLATFORM}")
set(CONTENT_DIR "${DIST_DIR}/UnityMeshSync/${MODO_YEAR}")

install(TARGETS ${MESHSYNC_CLIENT_MODO} DESTINATION "${CONTENT_DIR}/plug-ins/")
